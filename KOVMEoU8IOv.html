WIN SCRIPT

for /f "tokens=1-4 delims=-/ " %%i IN ("%date%") DO (

set year=%%i

set month=%%j

set day=%%k

)

for /f "tokens=1-4 delims=:." %%i IN ("%time%") DO (

set hour=%%i

set minute=%%j

set second=%%k

set centisecond=%%l

)

@echo %year%

@echo %month%

@echo %day%

@echo %hour%

@echo %minute%

@echo %second%

@echo %centisecond%

pause
